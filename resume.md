# 3. МОДЕЛЬ ДАННЫХ (БАЗА ДАННЫХ)
# ==================================================

# Таблица: projects
# ==================================================
# Хранит информацию о проектах (папка на диске)
CREATE TABLE projects (
    id          INT PRIMARY KEY AUTO_INCREMENT,          # Уникальный идентификатор
    name        VARCHAR(255) NOT NULL,                  # Название проекта
    folder_path VARCHAR(500) NOT NULL UNIQUE,           # Абсолютный путь к папке проекта
    created_at  TIMESTAMP DEFAULT NOW()                 # Дата создания
);


# Таблица: chats
# ==================================================
# Хранит диалоги (чаты) внутри проекта
CREATE TABLE chats (
    id          INT PRIMARY KEY AUTO_INCREMENT,          # Уникальный идентификатор
    project_id  INT NOT NULL,                           # Ссылка на проект
    title       VARCHAR(255) DEFAULT 'Новый чат',       # Название чата
    created_at  TIMESTAMP DEFAULT NOW(),                # Дата создания
    
    # Связи
    FOREIGN KEY (project_id) REFERENCES projects(id) ON DELETE CASCADE,
    
    # Индексы
    INDEX(project_id)
);


# Таблица: messages
# ==================================================
# Хранит историю переписки (контекст для ИИ)
CREATE TABLE messages (
    id            INT PRIMARY KEY AUTO_INCREMENT,       # Уникальный идентификатор
    chat_id       INT NOT NULL,                        # Ссылка на чат
    role          ENUM('user', 'assistant', 'system') NOT NULL,  # Роль отправителя
    content       TEXT NOT NULL,                       # Текст сообщения (чистый, без маркеров)
    
    # Статистика токенов
    input_tokens  INT NULL,                            # Количество входных токенов
    output_tokens INT NULL,                            # Количество выходных токенов
    total_tokens  INT NULL,                            # Общее количество токенов
    cost          DECIMAL(10, 6) NULL,                 # Стоимость запроса в USD
    
    created_at    TIMESTAMP DEFAULT NOW(),             # Дата создания
    
    # Связи
    FOREIGN KEY (chat_id) REFERENCES chats(id) ON DELETE CASCADE,
    
    # Индексы
    INDEX(chat_id),
    INDEX(created_at)
);


# Таблица: file_versions
# ==================================================
# Главная таблица. Хранит все версии всех файлов.
# Одна версия = одна запись.
CREATE TABLE file_versions (
    id            INT PRIMARY KEY AUTO_INCREMENT,       # Уникальный идентификатор
    chat_id       INT NOT NULL,                        # Ссылка на чат
    message_id    INT NULL,                            # Ссылка на сообщение (для generated)
    
    filename      VARCHAR(500) NOT NULL,               # Относительный путь внутри проекта
    content       LONGTEXT NOT NULL,                   # Содержимое файла
    content_hash  VARCHAR(64) NOT NULL,                # SHA-256 хеш содержимого
    
    file_type     ENUM('uploaded', 'generated', 'synced') NOT NULL,  # Тип файла
    language      VARCHAR(50) NULL,                    # Язык программирования (для подсветки)
    
    is_current    BOOLEAN DEFAULT FALSE,               # TRUE = сейчас на диске (одна версия на файл)
    applied       BOOLEAN DEFAULT FALSE,               # TRUE = одобрена пользователем
    
    created_at    TIMESTAMP DEFAULT NOW(),             # Дата создания
    
    # Связи
    FOREIGN KEY (chat_id) REFERENCES chats(id) ON DELETE CASCADE,
    FOREIGN KEY (message_id) REFERENCES messages(id) ON DELETE SET NULL,
    
    # Уникальность (защита от дублей)
    UNIQUE(chat_id, filename, content_hash),
    
    # Индексы
    INDEX(chat_id, filename),
    INDEX(chat_id, is_current),
    INDEX(content_hash)
);


# Таблица: snapshots
# ==================================================
# Хранит состояния проекта (линейная история с ветвлением)
CREATE TABLE snapshots (
    id              INT PRIMARY KEY AUTO_INCREMENT,     # Уникальный идентификатор
    project_id      INT NOT NULL,                      # Ссылка на проект
    
    prev_id         INT NULL,                          # Ссылка на предыдущее состояние (для ветвления)
    sequence_number INT NOT NULL,                      # Сквозной порядковый номер
    
    level           TINYINT NOT NULL DEFAULT 2,        # Уровень: 1-закреплённые, 2-основные, 3-автоматические
    type            ENUM('initial', 'manual', 'apply', 'upload', 'sync', 'auto') NOT NULL,
    
    name            VARCHAR(255) NULL,                 # Название (для уровней 1 и 2)
    description     TEXT NULL,                         # Описание (для уровней 1 и 2)
    
    files_manifest  JSON NOT NULL,                     # Список файлов: {"filename": "content_hash"}
    
    is_current      BOOLEAN DEFAULT FALSE,             # TRUE = текущее состояние (один на проект)
    created_at      TIMESTAMP DEFAULT NOW(),           # Дата создания
    
    # Связи
    FOREIGN KEY (project_id) REFERENCES projects(id) ON DELETE CASCADE,
    FOREIGN KEY (prev_id) REFERENCES snapshots(id) ON DELETE CASCADE,
    
    # Уникальность
    UNIQUE(project_id, sequence_number),
    
    # Индексы
    INDEX(project_id, is_current),
    INDEX(project_id, sequence_number),
    INDEX(prev_id)
);


# Таблица: system_prompts
# ==================================================
# Хранит системные промпты (предустановленные + пользовательские)
CREATE TABLE system_prompts (
    id          INT PRIMARY KEY AUTO_INCREMENT,        # Уникальный идентификатор
    name        VARCHAR(100) NOT NULL,                 # Название промпта
    content     TEXT NOT NULL,                         # Содержимое промпта
    
    is_default  BOOLEAN DEFAULT FALSE,                 # TRUE = промпт по умолчанию
    is_custom   BOOLEAN DEFAULT FALSE,                 # TRUE = пользовательский
    is_quick    BOOLEAN DEFAULT FALSE,                 # TRUE = доступен в быстром выборе
    
    created_at  TIMESTAMP DEFAULT NOW()                # Дата создания
);


# Таблица: usage_logs (опционально)
# ==================================================
# Хранит детальную статистику использования API
CREATE TABLE usage_logs (
    id            INT PRIMARY KEY AUTO_INCREMENT,       # Уникальный идентификатор
    chat_id       INT NOT NULL,                        # Ссылка на чат
    model         VARCHAR(50) NOT NULL,                # Использованная модель (flash/pro)
    input_tokens  INT NOT NULL,                        # Входные токены
    output_tokens INT NOT NULL,                        # Выходные токены
    cost          DECIMAL(10, 6) NOT NULL,             # Стоимость в USD
    created_at    TIMESTAMP DEFAULT NOW(),             # Дата создания
    
    # Связи
    FOREIGN KEY (chat_id) REFERENCES chats(id) ON DELETE CASCADE,
    
    # Индексы
    INDEX(chat_id),
    INDEX(created_at)
);

# 6. ПАРСИНГ ОТВЕТА ИИ
# ==================================================

# Функция parse_ai_response(content: str) -> Tuple[str, List[Dict]]
# Возвращает:
#   - text: текст без маркеров
#   - files: список словарей {type, filename, language, content}

# Типы:
#   - 'file': сохраняется в проект (маркер ### FILE:)
#   - 'snippet': пример кода (маркер ### CODE:)

# Определение языка:
#   1. Приоритет у языка, указанного моделью (если есть)
#   2. Если не указан → определяем по расширению файла
#   3. Если неизвестно → 'text'

# Безопасность:
#   - Защита от ../ (directory traversal)
#   - Только относительные пути
#   - Проверка размера файла (макс. 10MB)

# 17. ВЕРСИОНИРОВАНИЕ ФАЙЛОВ
# ==================================================

# Правила:
# ==================================================
# 1. Каждая версия файла — отдельная запись в file_versions
# 2. is_current = TRUE — только у одной версии на файл (то, что на диске)
# 3. applied = TRUE — пользователь сознательно одобрил эту версию

# ==================================================
# - Уровень 1: Закреплённые (initial + manual) — всегда видны
# - Уровень 2: Основные (apply + upload + sync) — видны по умолчанию
# - Уровень 3: Автоматические (auto) — скрыты (расширенный режим)

# Навигация:
# ==================================================
# - По времени: sequence_number (+1 / -1)
